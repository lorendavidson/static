const raw = window.parent.document.querySelector('.form-html pre').innerText; const contacts = JSON.parse(raw); const dropdown = window.parent.document.querySelector("select[name='wholesaleDistributors']"); dropdown.addEventListener('change', () => { const dist = contacts[dropdown.value]; if (dist) { window.parent.document.querySelector("input[name='wholesaleDistributorName']").value = dist.name; window.parent.document.querySelector("input[name='wholesaleDistributorEmail']").value = dist.email; } });
